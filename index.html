Sistema A — Imagen 2

% 8x1+4x2-x3=11 | -2x1+5x2+x3=4 | 2x1-x2+6x3=7
A = [ 8,  4, -1;
     -2,  5,  1;
      2, -1,  6];
b = [11; 4; 7];

[L,U,P] = lu(A);
y = L \ (P*b);
x = U \ y;
fprintf('x1=%.6f  x2=%.6f  x3=%.6f\n', x);
fprintf('Residuo = %.2e\n', norm(A*x-b));

Plantilla general N×N

% Modifique A y b según su sistema
A = [ ... ];
b = [ ... ];

%--- Directo ---
x = A \ b;

%--- LU con pivoteo ---
[L,U,P] = lu(A);  y = L\(P*b);  x_lu = U\y;

%--- Jacobi ---
n=length(b); x0=zeros(n,1); tol=1e-8; maxIt=500;
for k=1:maxIt
  xn=(b-(A-diag(diag(A)))*x0)./diag(A);
  if norm(xn-x0,inf)break; end; x0=xn;
end

%--- Gauss-Seidel ---
x0=zeros(n,1);
for k=1:maxIt
  xn=x0;
  for i=1:n
    xn(i)=(b(i)-A(i,:)*xn+A(i,i)*xn(i))/A(i,i);
  end
  if norm(xn-x0,inf)break; end; x0=xn;
end
